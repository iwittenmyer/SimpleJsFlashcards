function initAll() {
var rootFolder = new Folder("Flash Cards");
gDataList = [];

// First, create the special root folder
rootFolder.index = 0;
rootFolder.parent = null;
rootFolder.number = 0;

// Add this folder to the global list
gDataList.push(rootFolder);

gCurrentFolder = gDataList[0];

// Now fill with user-defined sub-folders and cards
initCards();

// Initialize global page info object
gPage = new Page();

//printItems();
writePage();
showPage();
}


function newFolder(name) {
var folder = new Folder(name);

// gDataList.length tells where this item is *going* to go
// parent.children.length tells where this item is *going* to go
folder.index = gDataList.length;
folder.parent = gCurrentFolder;

if (folder.parent.folderType == null) {

folder.parent.folderType = "folder";

} else if (folder.parent.folderType == "card") {

alert('Trying to add folder "' + name + '" to folder "' + folder.parent.name + '" but this folder already contains a card. Folders may only contain cards or other folders, but not both. Ignoring folder.');
return;
}

folder.number = folder.parent.children.length + 1;

// Add this folder to the global list
gDataList.push(folder);

// Add this folder to its parent's children
folder.parent.children.push(folder);

// Push down and make this the new current folder
gCurrentFolder = gDataList[folder.index];
}

function newSubject(name) {return newFolder(name);}
function newChapter(name) {return newFolder(name);}
function newSection(name) {return newFolder(name);}


function endFolder(name) {
if (name != gCurrentFolder.name) {
alert("Warning! You called endFolder with name '" + name +
"' but the currently open folder is '" +
gCurrentFolder.name + "'. Please check your file.");
} else {
// Pop up a folder
gCurrentFolder = gCurrentFolder.parent;
}
}

function endSubject(name) {return endFolder(name);}
function endChapter(name) {return endFolder(name);}
function endSection(name) {return endFolder(name);}


function addCard(frontText, backText) {
var card = new Card(frontText, backText);

card.index = gDataList.length;
card.parent = gCurrentFolder;

if (card.parent.folderType == null) {

card.parent.folderType = "card";

} else if (card.parent.folderType == "folder") {

alert('Trying to add card "' + frontText + '" to folder "' + card.parent.name + '" but this folder already contains another folder. Folders may only contain cards or other folders, but not both. Discarding card.');
return;
}

card.number = card.parent.children.length + 1;

// Add this card to the global list
gDataList.push(card);

// Add this folder to its parent's children
card.parent.children.push(card);
}


// Folder constructor
function Folder(name) {
// These properties can be set here in the constructor
this.type = "folder";
this.name = name;
this.children = [];

// These properties will be set later
this.index = null;
this.parent = null;
this.number = null;
this.folderType = null;
}


// Card constructor
function Card(frontText, backText) {
// These properties can be set here in the constructor
this.type = "card";
this.frontText = frontText;
this.backText = backText;

// These properties will be set later
this.index = null;
this.parent = null;
this.number = null;
}


// Page constructor
function Page() {

var myCookie;
var cookieSegments;
var result;

// Defaults

// Card index can show fronts, backs, or index (both)
// Default is index
this.cardIndexShow = "index";

// Card can show front or back side
// Default is front
this.cardSide = "front";

// Currently displayed index.
// Default is 0 (root)
this.pageIndex = 0; // Root

// Try to override defaults with cookie
myCookie = document.cookie;
if (!myCookie) {
return;
}

cookieSegments = myCookie.split(';');
for (i = 0; i < cookieSegments.length; i++) {
result = cookieSegments[i].match(/^\s*pageIndex=(.*)/);
if (result && result[1]) {
this.pageIndex = result[1];
}

result = cookieSegments[i].match(/^\s*cardIndexShow=(.*)/);
if (result && result[1]) {
this.cardIndexShow = result[1];
}

result = cookieSegments[i].match(/^\s*cardSide=(.*)/);
if (result && result[1]) {
this.cardSide = result[1];
}
}
}


function printItems() {
var text;
var i;
var myItem;

for (i = 0; i < gDataList.length; i++) {
myItem = gDataList[i];
if (myItem.type == "folder") {
text = myItem.index + " Folder: " + myItem.name + "\n";
} else if (myItem.type == "card") {
text = myItem.index + " Card: " + myItem.frontText + "-->" + myItem.backText + "\n";
} else {
text = myItem.index + "Error!\n";
}
document.write(text);
}
}

function pageItem() {
return gDataList[gPage.pageIndex];
}


function genNavPath(element) {
var navPath = "";
if (element.type == "card") {
element = element.parent;
}
while (true) {
navPath = '/' + '<a class=a_nav_segment href=# onclick="return handleClickA(' + element.index + ')">' + element.name + '</a>' + navPath;
if (element.parent) {
element = element.parent;
} else {
break;
}
}

return navPath;
}


function genNavBar(element) {
var navBar = "<div class=navbar>";
navBar = navBar + "<a class=a_home href=\"index.htm\" title=\"Go to site's home\" target=_top rel=home>home</a> | ";
if (element.index != 0) {
navBar = navBar + "<a class=a_up href=# onclick=\"return handleClickA(" + element.parent.index + ")\" title=\"Go up to containing folder\" rel=parent rev=child>up</a> | ";
}
navBar = navBar + "<span class=navpath>" + genNavPath(element) + "</span>";
if (element.type == "card") {
navBar = navBar + " <span class=card_counter>card " + element.number + " of " + element.parent.children.length + "</span>";
}

// if (element.index == 0) {
// document.write("<br>BROWSER INFORMATION:");
// for(var propname in navigator) {
// document.write("<br>", propname, ": ", navigator[propname]);
// }
// }

navBar = navBar + "<!--navbar--></div>"
return navBar;
}

function showPage() {
var j;
var myItem;

myItem = pageItem();

if (myItem.type == "folder") {
document.body.style.backgroundColor = "white";
if (myItem.folderType == "folder") {
gElements.folder_index_template.style.visibility = "visible";
document.getElementById("folder_index_" + myItem.index).style.visibility = "visible";
} else if (myItem.folderType == "card") {
gElements.card_index_template.style.visibility = "visible";

document.getElementById("card_index_" + myItem.index).style.visibility = "visible";

if (gPage.cardIndexShow == "fronts") {
document.getElementById("div_fronts_" + myItem.index).style.visibility = "visible";
document.getElementById("form_show_fronts_" + myItem.index).checked = "true";
} else if (gPage.cardIndexShow == "backs") {
document.getElementById("div_backs_" + myItem.index).style.visibility = "visible";
document.getElementById("form_show_backs_" + myItem.index).checked = "true";
} else {
document.getElementById("div_index_" + myItem.index).style.visibility = "visible";
document.getElementById("form_show_index_" + myItem.index).checked = "true";
}
} else {
alert("Error! showPage Unknown folderType: " + myItem.folderType);
}
} else if (myItem.type == "card") {
document.body.style.backgroundColor = "silver";
gElements.card_template.style.visibility = "visible";
if (gPage.cardSide == "back") {
gElements.form_side_front.checked = false;
gElements.form_side_back.checked = true;
gElements.card_back.style.visibility = "visible";
gElements.card_back_text[myItem.index].style.visibility = "visible";
} else {
gElements.form_side_back.checked = false;
gElements.form_side_front.checked = true;
gElements.card_front.style.visibility = "visible";
gElements.card_front_text[myItem.index].style.visibility = "visible";
}
gElements.card_nav[myItem.index].style.visibility = "visible";
}
window.scrollTo(0,0); // Scroll to the top of the window

document.cookie = "pageIndex=" + gPage.pageIndex;
document.cookie = "cardIndexShow=" + gPage.cardIndexShow;
document.cookie = "cardSide=" + gPage.cardSide;

if ((myItem.type == "card") && (gPage.cardSide == "back")) {
// Due to a PSP browser bug, need to poke table style to refresh border color
// Plus, do this last because it's reeeeeeally slow
// gElements.table_rules.style.borderStyle = "none";
}
}

function handleClickA(nextIndex, option) {
var myItem = pageItem();
// Hide sub elements in current item
if (myItem.type == "folder") {
if (myItem.folderType == "folder") {
document.getElementById("folder_index_" + gPage.pageIndex).style.visibility = "hidden";
if ((gDataList[nextIndex].type != "folder") || (gDataList[nextIndex].folderType != "folder")) {
gElements.folder_index_template.style.visibility = "hidden";
}
} else if (myItem.folderType == "card") {
document.getElementById("card_index_" + gPage.pageIndex).style.visibility = "hidden";
document.getElementById("div_index_" + gPage.pageIndex).style.visibility = "hidden";
document.getElementById("div_fronts_" + gPage.pageIndex).style.visibility = "hidden";
document.getElementById("div_backs_" + gPage.pageIndex).style.visibility = "hidden";
if ((gDataList[nextIndex].type != "folder") || (gDataList[nextIndex].folderType != "card")) {
gElements.card_index_template.style.visibility = "hidden";
}
}
} else if (myItem.type == "card") {
gElements.card_front_text[gPage.pageIndex].style.visibility = "hidden";
gElements.card_back_text[gPage.pageIndex].style.visibility = "hidden";
gElements.card_nav[gPage.pageIndex].style.visibility = "hidden";
gElements.card_front.style.visibility = "hidden";
if (gDataList[nextIndex].type != "card") {
gElements.card_back.style.visibility = "hidden";
gElements.card_template.style.visibility = "hidden";
}
}

// Switch to new page

gPage.pageIndex = nextIndex;

if ((option == "fronts") || (option == "backs") || (option == "index")) {
gPage.cardIndexShow = option;
} else if ((option == "front") || (option == "back")) {
gPage.cardSide = option;
}

showPage();
return false;
}


function handleClickAPrev() {
var myItem = pageItem();
var prevIndex;
if (myItem.number > 1) {
prevIndex = myItem.index - 1;
} else {
prevIndex = myItem.parent.index;
}
return handleClickA(prevIndex);
}


function handleClickANext() {
var myItem = pageItem();
var nextIndex;
if (myItem.number < myItem.parent.children.length) {
nextIndex = myItem.index + 1;
} else {
nextIndex = myItem.parent.index;
}
return handleClickA(nextIndex);
}

function handleClickAStep() {
if (gPage.cardSide == "back") {
gPage.cardSide = "front";
handleClickANext();
} else {
gPage.cardSide = "back";
gElements.card_front_text[gPage.pageIndex].style.visibility = "hidden";
gElements.card_front.style.visibility = "hidden";
showPage();
}

return false;
}

function handleClickFormSide(mythis) {
switch (mythis.value) {
case "back":
gPage.cardSide = "back";
break;
case "front":
default:
gPage.cardSide = "front";
}

gElements.card_front_text[gPage.pageIndex].style.visibility = "hidden";
// gElements.card_back_text[gPage.pageIndex].style.visibility = "hidden";
gElements.card_nav[gPage.pageIndex].style.visibility = "hidden";
gElements.card_front.style.visibility = "hidden";
gElements.card_back.style.visibility = "hidden";

showPage();

// Returning true allows the browser to animate the button
return true;
}
